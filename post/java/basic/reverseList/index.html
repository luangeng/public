反转单向链表

Posted @ 2016-09-03 · 300 Words · Tags: 算法 · Hits:

1.请先实现一个单向链表

public class MyLinkedList {

    private static class Node {
        int value;
        Node next;
        public Node(int value) {
            this.value = value;
        }
    }

    private Node head;

    private int size = 0;

    //FILO
    public void push(int v) {
        size++;
        if (head == null) {
            head = new Node(v);
        } else {
            Node node = new Node(v);
            node.next = head;
            head = node;
        }
    }

    public Node pop() {
        Node node = head;
        if (node == null) {
            return null;
        }
        head = head.next;
        size--;
        return node;
    }

    public int size() {
        return size;
    }

    public void print() {
        StringBuilder sb = new StringBuilder();
        Node node = head;
        if (node == null) {
            return;
        }
        do {
            sb.append(node.value);
            sb.append(' ');
            node = node.next;
        } while (node != null);
        System.out.println(sb.toString());
    }
}

测试

    public static void main(String[] args) {

        MyLinkedList list = new MyLinkedList();

        for (int k = 1; k <= 15; k++) {
            list.push(k);
        }

        list.print();
   }

再写一个方法反转链表

    //反转单向链表
    public void reverse() {
        if (head == null) {
            return;
        }
        Node next = head.next;
        head.next = null;
        do {
            Node pre = head;
            head = next;
            next = next.next;
            head.next = pre;
        } while (next != null);
    }

测试

 public static void main(String[] args) {

        MyLinkedList list = new MyLinkedList();

        for (int k = 1; k <= 15; k++) {
            list.push(k);
        }

        list.print();
        list.reverse();
        list.print();

    }
